var Model = function() {
  var AllCards = 52;
  var KingCards = 4;
  var ElseCards = AllCards - KingCards;

  var KingThenElse = (KingCards / AllCards) * (ElseCards / (AllCards -1));
  var ElseThenKing = (ElseCards / AllCards) * (KingCards / (AllCards -1));
  
  var TotalP = (KingThenElse + ElseThenKing);
  
  return TotalP;
};

print("The probability of choosing one King and one Non-king once: " + Model().toFixed(4));
